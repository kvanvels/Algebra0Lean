\section{Definition of group}
\label{sec:groups:definition-of-group}

\begin{definition}[Group]
  \label{def:groups:group}
  \lean{Algebra0Lean.Groups.Group}
  \leanok
  The set $G$, endowed with a binary operation $\bullet$, is a
  \emph{group} if $\bullet$ is associative, has an identity element
  $e_G$, and every element has an inverse.
\end{definition}

\begin{definition}[Identity element]
  \label{def:groups:is-identity}
  \uses{def:groups:group}
  \lean{Algebra0Lean.Groups.IsIdentity}
  \leanok
  An element $h \in G$ is an identity for the group law if
  $gh = g = hg$ for all $g \in G$.
\end{definition}

\begin{proposition}[II.1.2]
  \label{prop:groups:identity-unique}
  \uses{def:groups:is-identity}
  \lean{Algebra0Lean.Groups.isIdentity_eq}
  \leanok
  If $h \in G$ is an identity of $G$, then $h = e_G$.
\end{proposition}
\begin{proof}
  \leanok
\end{proof}

\begin{definition}[Inverse]
  \label{def:groups:is-inverse}
  \uses{def:groups:group}
  \lean{Algebra0Lean.Groups.IsInverse}
  \leanok
  An element $h \in G$ is an inverse of $g \in G$ if
  $gh = e_G = hg$.
\end{definition}

\begin{proposition}[II.1.3]
  \label{prop:groups:inverse-unique}
  \uses{def:groups:is-inverse}
  \lean{Algebra0Lean.Groups.isInverse_unique}
  \leanok
  The inverse is also unique: if $h_1$, $h_2$ are both inverses of $g$
  in $G$, then $h_1 = h_2$.
\end{proposition}
\begin{proof}
  \leanok
\end{proof}

\subsection{Cancellation}
\label{subsec:groups:cancellation}

\begin{proposition}[II.1.4, cancellation]
  \label{prop:groups:cancellation}
  \uses{def:groups:group}
  \lean{Algebra0Lean.Groups.cancel_left, Algebra0Lean.Groups.cancel_right}
  \leanok
  Let $G$ be a group. Then $\forall a, g, h \in G$
  \[
    ga = ha \implies g = h, \qquad ag = ah \implies g = h.
  \]
\end{proposition}
\begin{proof}
  \leanok
\end{proof}

\subsection{Commutative groups}

\begin{definition}[Commutative group]
  \label{def:groups:is-commutative}
  \uses{def:groups:group}
  \lean{Algebra0Lean.Groups.IsCommutative}
  \leanok
  The operation $\bullet$ is \emph{commutative} if $gh = hg$ for all
  $g, h \in G$. A group whose operation is commutative is called
  \emph{abelian}.
\end{definition}

\subsection{Order}
\label{subsec:groups:order}

\begin{definition}[Powers]
  \label{def:groups:powers}
  \uses{def:groups:group}
  \lean{Algebra0Lean.Groups.gpow, Algebra0Lean.Groups.gzpow}
  \leanok
  For $g \in G$ and $n \geq 0$, the power $g^n = g \cdots g$ ($n$
  times), with $g^0 = e_G$; for negative exponents,
  $g^{-n} = (g^n)^{-1}$.
\end{definition}

\begin{proposition}
  \label{prop:groups:gpow-add}
  \uses{def:groups:powers}
  \lean{Algebra0Lean.Groups.gpow_add}
  \leanok
  Powers add: for $g \in G$ and $a, b \geq 0$, $g^{a+b} = g^a g^b$.
\end{proposition}
\begin{proof}
  \leanok
\end{proof}

\begin{proposition}
  \label{prop:groups:gzpow-add}
  \uses{def:groups:powers}
  \lean{Algebra0Lean.Groups.gzpow_add}
  For $g \in G$ and $a, b \in \Z$, $g^{a+b} = g^a g^b$: the book's
  stated form of the power-addition law, for all integer exponents
  (not just $a, b \geq 0$).
\end{proposition}

\begin{proposition}
  \label{prop:groups:gpow-mul}
  \uses{def:groups:powers}
  \lean{Algebra0Lean.Groups.gpow_mul}
  \leanok
  Powers multiply: for $g \in G$ and $a, b \geq 0$, $g^{ab} = (g^a)^b$.
\end{proposition}
\begin{proof}
  \uses{prop:groups:gpow-add}
  \leanok
\end{proof}

\begin{proposition}
  \label{prop:groups:gpow-e}
  \uses{def:groups:powers}
  \lean{Algebra0Lean.Groups.gpow_e}
  \leanok
  Every power of the identity is the identity: $e_G^n = e_G$ for all
  $n \geq 0$.
\end{proposition}
\begin{proof}
  \leanok
\end{proof}

\begin{proposition}
  \label{prop:groups:gpow-op-comm}
  \uses{def:groups:powers}
  \lean{Algebra0Lean.Groups.gpow_op_comm}
  \leanok
  If $h$ commutes with $g$ (i.e.\ $gh = hg$), then $h$ commutes with
  every power of $g$: $g^n h = h g^n$ for all $n \geq 0$.
\end{proposition}
\begin{proof}
  \leanok
\end{proof}

\begin{proposition}
  \label{prop:groups:gpow-op-of-comm}
  \uses{def:groups:powers}
  \lean{Algebra0Lean.Groups.gpow_op_of_comm}
  \leanok
  If $g$ and $h$ commute, powers distribute over the product:
  $(gh)^n = g^n h^n$ for all $n \geq 0$.
\end{proposition}
\begin{proof}
  \uses{prop:groups:gpow-op-comm}
  \leanok
\end{proof}

\begin{proposition}
  \label{prop:groups:inv-e}
  \uses{def:groups:group}
  \lean{Algebra0Lean.Groups.inv_e}
  \leanok
  The identity is its own inverse: $e_G^{-1} = e_G$.
\end{proposition}
\begin{proof}
  \leanok
\end{proof}

\begin{proposition}
  \label{prop:groups:inv-eq-e-iff}
  \uses{def:groups:group}
  \lean{Algebra0Lean.Groups.inv_eq_e_iff}
  \leanok
  For $x \in G$: $x^{-1} = e_G$ if and only if $x = e_G$.
\end{proposition}
\begin{proof}
  \uses{prop:groups:inv-e}
  \leanok
\end{proof}

\begin{definition}[II.1.5, order of an element]
  \label{def:groups:order-element}
  \uses{def:groups:powers}
  \lean{Algebra0Lean.Groups.HasFiniteOrder, Algebra0Lean.Groups.order}
  \leanok
  An element $g \in G$ has \emph{finite order} if $g^n = e_G$ for some
  positive integer $n$. In this case, the \emph{order} $|g|$ is the
  smallest positive $n$ such that $g^n = e_G$. One writes
  $|g| = \infty$ if $g$ does not have finite order (in the Lean
  formalization, $|g| = 0$).
\end{definition}

\begin{proposition}
  \label{prop:groups:order-pos}
  \uses{def:groups:order-element}
  \lean{Algebra0Lean.Groups.order_pos}
  \leanok
  An element of finite order has positive order.
\end{proposition}
\begin{proof}
  \leanok
\end{proof}

\begin{proposition}
  \label{prop:groups:gpow-order}
  \uses{def:groups:order-element}
  \lean{Algebra0Lean.Groups.gpow_order}
  \leanok
  For an element $g$ of finite order, $g^{|g|} = e_G$.
\end{proposition}
\begin{proof}
  \leanok
\end{proof}

\begin{proposition}
  \label{prop:groups:gpow-ne-e-of-lt-order}
  \uses{def:groups:order-element}
  \lean{Algebra0Lean.Groups.gpow_ne_e_of_lt_order}
  \leanok
  Minimality of the order: for an element $g$ of finite order, no
  smaller positive power of $g$ is $e_G$, i.e.\ $g^m \neq e_G$ for
  $0 < m < |g|$.
\end{proposition}
\begin{proof}
  \leanok
\end{proof}

\begin{proposition}
  \label{prop:groups:order-eq-zero-of-infinite}
  \uses{def:groups:order-element}
  \lean{Algebra0Lean.Groups.order_eq_zero_of_infinite}
  \leanok
  An element of infinite order has order $0$ (Aluffi's $|g| = \infty$).
\end{proposition}
\begin{proof}
  \leanok
\end{proof}

\begin{proposition}
  \label{prop:groups:has-finite-order-of-order-pos}
  \uses{def:groups:order-element}
  \lean{Algebra0Lean.Groups.hasFiniteOrder_of_order_pos}
  \leanok
  Conversely, an element of positive order has finite order.
\end{proposition}
\begin{proof}
  \uses{prop:groups:order-eq-zero-of-infinite}
  \leanok
\end{proof}

\begin{lemma}[II.1.5]
  \label{lem:groups:order-divisor}
  \uses{def:groups:order-element}
  \lean{Algebra0Lean.Groups.order_dvd_of_pow_eq_e}
  \leanok
  If $g^n = e_G$ for some positive integer $n$, then $|g|$ is a
  divisor of $n$.
\end{lemma}
\begin{proof}
  \uses{prop:groups:order-pos, prop:groups:gpow-ne-e-of-lt-order}
  \leanok
\end{proof}

\begin{proposition}
  \label{prop:groups:gpow-eq-e-iff-order-dvd}
  \uses{def:groups:order-element}
  \lean{Algebra0Lean.Groups.gpow_eq_e_iff_order_dvd}
  \leanok
  Restatement of Lemma II.1.5 as an equivalence, for natural exponents:
  for an element $g$ of finite order, $g^n = e_G$ if and only if $|g|$
  divides $n$.
\end{proposition}
\begin{proof}
  \uses{lem:groups:order-divisor}
  \leanok
\end{proof}

\begin{corollary}[II.1.6]
  \label{coro:groups:order-multiple}
  \uses{def:groups:order-element}
  \lean{Algebra0Lean.Groups.gzpow_eq_e_iff_dvd_order}
  \leanok
  Let $g$ be an element of finite order, and let $N \in \mathbb{Z}$.
  Then
  \[
    g^N = e_G \iff N \text{ is a multiple of } |g|.
  \]
\end{corollary}
\begin{proof}
  \uses{lem:groups:order-divisor, prop:groups:gpow-eq-e-iff-order-dvd}
  \leanok
\end{proof}

\begin{definition}[Order of a group]
  \label{def:groups:order-group}
  \lean{Algebra0Lean.Groups.groupOrder}
  \leanok
  If $G$ is finite as a set, its \emph{order} $|G|$ is the number of
  its elements; $|G| = \infty$ if $G$ is infinite (in the Lean
  formalization, $|G| = 0$).
\end{definition}

\begin{proposition}[II.1.7]
  \label{prop:groups:order-power}
  \uses{def:groups:order-element}
  \lean{Algebra0Lean.Groups.order_gpow}
  \leanok
  Let $g \in G$ be an element of finite order. Then $g^m$ has finite
  order for all $m > 0$, and
  \[
    |g^m| = \frac{\operatorname{lcm}(m,|g|)}{m} =
    \frac{|g|}{\gcd(m,|g|)}.
  \]
\end{proposition}
\begin{proof}
  \uses{lem:groups:order-divisor, coro:groups:order-multiple}
  \leanok
\end{proof}

\begin{proposition}[II.1.8]
  \label{prop:groups:order-product}
  \uses{def:groups:order-element}
  \lean{Algebra0Lean.Groups.order_op_dvd_lcm}
  \leanok
  If $gh = hg$, then $|gh|$ divides $\operatorname{lcm}(|g|,|h|)$.
\end{proposition}
\begin{proof}
  \uses{lem:groups:order-divisor, coro:groups:order-multiple}
  \leanok
\end{proof}

\subsection{Symmetric groups}

\begin{definition}[Symmetric group]
  \label{def:groups:symmetric-group}
  \uses{def:groups:group, def:prelims:aut, def:prelims:type-category}
  \lean{Algebra0Lean.Groups.symmetricGroup}
  Let $A$ be a set. The \textbf{symmetric group}, or \textbf{group of
    permutations}, of $A$, denoted $S_A$, is the group
  $\Aut_{\categ{Set}}(A)$: its elements are the bijections
  $A \to A$, with composition as the group law.
\end{definition}

\subsection{Cyclic groups and modular arithmetic}
\label{subsec:groups:cyclic-groups-and-modular-arithmetic}

\begin{definition}[Congruence mod $n$]
  \label{def:groups:cong-mod}
  \lean{Algebra0Lean.Groups.congMod}
  \leanok
  For $a, b \in \Z$: $a \equiv b \pmod{n}$ iff $n$ divides $b - a$.
\end{definition}

\begin{proposition}
  \label{prop:groups:equivalence-cong-mod}
  \uses{def:groups:cong-mod}
  \lean{Algebra0Lean.Groups.equivalence_congMod}
  Congruence modulo $n$ is an equivalence relation on $\Z$.
\end{proposition}

\begin{lemma}
  \label{lem:groups:congruence-addition}
  \uses{def:groups:cong-mod}
  \lean{Algebra0Lean.Groups.congMod_add}
  \leanok
  If $a \equiv a' \pmod{n}$ and $b \equiv b' \pmod{n}$, then
  $(a+b) \equiv (a'+b') \pmod{n}$.
\end{lemma}
\begin{proof}
  \leanok
\end{proof}

\begin{definition}[$\Z/n\Z$]
  \label{def:groups:zmod-group}
  \uses{def:groups:group, prop:groups:equivalence-cong-mod,
    lem:groups:congruence-addition}
  \lean{Algebra0Lean.Groups.zmodSetoid, Algebra0Lean.Groups.zmodGroup}
  The (additive) group $\Z/n\Z$ of congruence classes mod $n$: the
  quotient of $\Z$ by congruence mod $n$, with
  $[a]_n + [b]_n := [a+b]_n$.
\end{definition}

\begin{proposition}
  \label{prop:groups:order-class}
  \uses{def:groups:zmod-group, def:groups:order-element}
  \lean{Algebra0Lean.Groups.order_class}
  The order of $[m]_n$ in $\Z/n\Z$ is $1$ if $n \mid m$, and more
  generally $|[m]_n| = \dfrac{n}{\gcd(m,n)}$.
\end{proposition}

\begin{definition}[Generator]
  \label{def:groups:generates}
  \uses{def:groups:group, def:groups:powers}
  \lean{Algebra0Lean.Groups.Generates}
  \leanok
  An element $g \in G$ \textbf{generates} $G$ if every element of $G$
  is an integer power of $g$.
\end{definition}

\begin{corollary}
  \label{coro:groups:generator-coprime}
  \uses{def:groups:zmod-group, def:groups:generates}
  \lean{Algebra0Lean.Groups.generates_iff_coprime}
  The class $[m]_n$ generates $\Z/n\Z$ if and only if $\gcd(m,n) = 1$.
\end{corollary}
\begin{proof}
  \uses{prop:groups:order-class}
\end{proof}

\begin{lemma}
  \label{lem:groups:multiplication-well-defined}
  \uses{def:groups:cong-mod}
  \lean{Algebra0Lean.Groups.congMod_mul}
  If $a \equiv a' \pmod{n}$ and $b \equiv b' \pmod{n}$, then
  $ab \equiv a'b' \pmod{n}$: multiplication $[a]_n \cdot [b]_n := [ab]_n$
  is well-defined on $\Z/n\Z$.
\end{lemma}

\begin{lemma}
  \label{lem:groups:coprime-class-well-defined}
  \uses{def:groups:cong-mod}
  \lean{Algebra0Lean.Groups.gcd_eq_one_congMod}
  Coprimality with $n$ only depends on the congruence class mod $n$:
  if $m \equiv m' \pmod n$, then $\gcd(m,n) = 1 \iff \gcd(m',n) = 1$.
\end{lemma}

\begin{definition}[Multiplication on $\Z/n\Z$]
  \label{def:groups:zmod-mul}
  \uses{def:groups:zmod-group, lem:groups:multiplication-well-defined}
  \lean{Algebra0Lean.Groups.zmodMul}
  Multiplication of congruence classes mod $n$:
  $[a]_n \cdot [b]_n := [ab]_n$.
\end{definition}

\begin{definition}[$(\Z/n\Z)^*$]
  \label{def:groups:zmod-units}
  \uses{def:groups:zmod-group, lem:groups:coprime-class-well-defined}
  \lean{Algebra0Lean.Groups.isUnitClass, Algebra0Lean.Groups.zmodUnits}
  For any positive $n$, denote by $(\Z/n\Z)^*$ the subset of $\Z/n\Z$
  consisting of classes $[m]_n$ such that $\gcd(m,n)=1$:
  \[
    (\Z/n\Z)^* := \{[m]_n \in \Z/n\Z \mid \gcd(m,n)=1\}.
  \]
\end{definition}

\begin{proposition}
  \label{prop:groups:units-group}
  \uses{def:groups:zmod-mul, def:groups:zmod-units}
  \lean{Algebra0Lean.Groups.isUnitClass_mul, Algebra0Lean.Groups.zmodUnitsGroup}
  Multiplication makes $(\Z/n\Z)^*$ into a group.
\end{proposition}
\begin{proof}
  \uses{coro:groups:generator-coprime}
\end{proof}

\begin{definition}[Group homomorphism]
  \label{def:groups:homomorphism}
  \uses{def:groups:group}
  \lean{Algebra0Lean.Groups.IsGroupHom}
  \leanok
  For groups $(G, m_G)$ and $(H, m_H)$, a set-function
  $\varphi : G \to H$ is a \textbf{group homomorphism} if
  $\varphi(a \cdot b) = \varphi(a) \cdot \varphi(b)$ for all
  $a, b \in G$.
\end{definition}

\subsection{Pause for reflection}
\label{subsec:groups:pause-for-reflection}

\begin{proposition}
  \label{prop:groups:homo-preserves-identity}
  \uses{def:groups:homomorphism}
  \lean{Algebra0Lean.Groups.IsGroupHom.map_e,
        Algebra0Lean.Groups.IsGroupHom.map_inv}
  Let $\varphi : G \to H$ be a group homomorphism. Then
  $\varphi(e_G) = e_H$, and $\forall g \in G$,
  $\varphi(g^{-1}) = \varphi(g)^{-1}$.
\end{proposition}

\subsection{Homomorphisms of abelian groups}
\label{subsec:groups:homomorphisms-of-abelian-groups}

\begin{definition}
  \label{def:groups:group-hom-type}
  \uses{def:groups:homomorphism}
  \lean{Algebra0Lean.Groups.GroupHom}
  \leanok
  For groups $G$, $H$, we write $\Hom(G, H)$ for the type of group
  homomorphisms $G \to H$.
\end{definition}

\begin{proposition}
  \label{prop:groups:hom-abelian-group}
  \uses{def:groups:group-hom-type, def:groups:is-commutative}
  \lean{Algebra0Lean.Groups.homGroup}

  $(\varphi + \psi)(g) := \varphi(g) \psi(g)$ defined pointwise, is

\begin{proposition}

\end{proposition}
\begin{proof}
  \uses{prop:groups:hom-abelian-group}
\end{proof}

\section{Free groups}
\label{sec:groups:free-groups}

\subsection{Universal property}
\label{subsec:groups:free-groups-universal-property}

\begin{definition}
  \label{def:groups:is-free-group-on}
  \uses{def:groups:group, def:groups:homomorphism}
  \lean{Algebra0Lean.Groups.IsFreeGroupOn}
  \leanok
  A group $F$, together with a function $\iota : S \to F$, is the
  \emph{free group on $S$} if for every group $H$ and every function
  $f : S \to H$, there is a unique group homomorphism $\varphi : F
  \to H$ such that $\varphi \circ \iota = f$.
\end{definition}

\begin{proposition}
  \label{prop:groups:free-group-unique}
  \uses{def:groups:is-free-group-on, def:groups:isomorphic-groups}
  \lean{Algebra0Lean.Groups.isomorphicGroups_of_isFreeGroupOn}
  If $(F_1, \iota_1)$ and $(F_2, \iota_2)$ are both free groups on
  $S$, then $F_1 \cong F_2$.
\end{proposition}

\end{definition}

\begin{proposition}

\section{Subgroups}
\label{sec:groups:subgroups}

\subsection{Definition}
\label{subsec:groups:subgroups-definition}

\begin{definition}
  \label{def:groups:is-subgroup}
  \uses{def:groups:group}
  \lean{Algebra0Lean.Groups.IsSubgroup}
  \leanok
  A subset $H$ of $G$ is a \emph{subgroup} of $G$ if it contains the
  identity and is closed under the operation and under taking
  inverses.
\end{definition}

\subsection{Kernel and image}
\label{subsec:groups:kernel-and-image}

\begin{definition}
  \label{def:groups:kernel}
  \uses{def:groups:homomorphism}
  \lean{Algebra0Lean.Groups.GroupHom.ker}
  \leanok
  The \emph{kernel} of a group homomorphism $\varphi : G \to H$ is
  $\ker\varphi := \{g \in G \mid \varphi(g) = e_H\}$.
\end{definition}

\begin{definition}
  \label{def:groups:image}
  \uses{def:groups:homomorphism}
  \lean{Algebra0Lean.Groups.GroupHom.image}
  \leanok
  The \emph{image} of a group homomorphism $\varphi : G \to H$ is
  $\operatorname{im}\varphi := \{\varphi(g) \mid g \in G\}$.
\end{definition}

\begin{proposition}
  \label{prop:groups:kernel-is-subgroup}
  \uses{def:groups:is-subgroup, def:groups:kernel}
  \lean{Algebra0Lean.Groups.isSubgroup_ker}
  The kernel of a group homomorphism is a subgroup of the domain.
\end{proposition}

\begin{proposition}
  \label{prop:groups:image-is-subgroup}
  \uses{def:groups:is-subgroup, def:groups:image}
  \lean{Algebra0Lean.Groups.isSubgroup_image}
  The image of a group homomorphism is a subgroup of the codomain.
\end{proposition}

\subsection{Monomorphisms}
\label{subsec:groups:monomorphisms}

\begin{proposition}
  \label{prop:groups:injective-iff-trivial-kernel}
  \uses{def:groups:kernel}
  \lean{Algebra0Lean.Groups.injective_iff_ker_eq_singleton_e}
  A group homomorphism $\varphi : G \to H$ is injective if and only
  if $\ker\varphi = \{e_G\}$.
\end{proposition}

\subsection{Subgroup generated by a subset}
\label{subsec:groups:generated-subgroup}

\begin{definition}
  \label{def:groups:generated-subgroup}
  \uses{def:groups:is-subgroup}
  \lean{Algebra0Lean.Groups.generatedSubgroup}
  \leanok
  The \emph{subgroup generated by} a subset $S$ of $G$, written
  $\langle S \rangle$, is the intersection of all subgroups of $G$
  containing $S$.
\end{definition}

\begin{proposition}
  \label{prop:groups:generated-subgroup-is-subgroup}
  \uses{def:groups:generated-subgroup}
  \lean{Algebra0Lean.Groups.isSubgroup_generatedSubgroup}
  $\langle S \rangle$ is a subgroup of $G$.
\end{proposition}

\begin{proposition}
  \label{prop:groups:subset-generated-subgroup}
  \uses{def:groups:generated-subgroup}
  \lean{Algebra0Lean.Groups.subset_generatedSubgroup}
  $S \subseteq \langle S \rangle$.
\end{proposition}

\begin{proposition}
  \label{prop:groups:generated-subgroup-minimal}
  \uses{def:groups:generated-subgroup, def:groups:is-subgroup}
  \lean{Algebra0Lean.Groups.generatedSubgroup_subset}
  $\langle S \rangle$ is the smallest subgroup of $G$ containing $S$:
  if $H$ is a subgroup with $S \subseteq H$, then $\langle S \rangle
  \subseteq H$.
\end{proposition}

\subsection{Subgroups of cyclic groups}
\label{subsec:groups:subgroups-of-cyclic-groups}

\begin{definition}
  \label{def:groups:subgroup-to-group}
  \uses{def:groups:is-subgroup}
  \lean{Algebra0Lean.Groups.IsSubgroup.toGroup}
  \leanok
  A subgroup $H$ of $G$, with the group structure inherited from $G$.

\begin{proposition}

  \uses{def:groups:subgroup-to-group, def:groups:is-cyclic-group}
  \lean{Algebra0Lean.Groups.isCyclicGroup_of_isSubgroup_of_isCyclicGroup}

\section{Quotient groups}
\label{sec:groups:quotient-groups}

\subsection{Normal subgroups}
\label{subsec:groups:normal-subgroups}

\begin{definition}
  \label{def:groups:normal-subgroup}
  \uses{def:groups:is-subgroup}
  \lean{Algebra0Lean.Groups.IsNormalSubgroup}
  \leanok
  A subgroup $N$ of a group $G$ is \emph{normal} if $\forall g \in G$,
  $\forall n \in N$, $gng^{-1} \in N$.
\end{definition}

\begin{proposition}
  \label{prop:groups:kernel-is-normal}
  \uses{def:groups:normal-subgroup, def:groups:kernel}
  \lean{Algebra0Lean.Groups.isNormalSubgroup_ker}
  If $\varphi : G \to G'$ is any group homomorphism, then
  $\ker\varphi$ is a normal subgroup of $G$.
\end{proposition}
\begin{proof}
  \uses{prop:groups:kernel-is-subgroup}
\end{proof}

\subsection{Cosets}
\label{subsec:groups:cosets}

\begin{definition}
  \label{def:groups:cosets}
  \uses{def:groups:group}
  \lean{Algebra0Lean.Groups.leftCoset, Algebra0Lean.Groups.rightCoset}
  \leanok
  The \emph{left-cosets} of a subgroup $H$ in a group $G$ are the sets
  $aH$, for $a \in G$. The \emph{right-cosets} of $H$ are the sets
  $Ha$, $a \in G$.
\end{definition}

\subsection{Quotient by normal subgroups}
\label{subsec:groups:quotient-by-normal-subgroups}

\begin{definition}
  \label{def:groups:coset-rel}
  \uses{def:groups:normal-subgroup}
  \lean{Algebra0Lean.Groups.cosetRel}
  \leanok
  The equivalence relation $a \sim b \iff a^{-1}b \in N$ corresponding
  to a (normal) subgroup $N$, whose classes are the left cosets of
  $N$.
\end{definition}

\begin{proposition}
  \label{prop:groups:equivalence-coset-rel}
  \uses{def:groups:coset-rel}
  \lean{Algebra0Lean.Groups.equivalence_cosetRel}
  For $N$ a normal subgroup, $\sim$ is an equivalence relation.
\end{proposition}

\begin{definition}
  \label{def:groups:quotient-group-modulo-n}
  \uses{def:groups:normal-subgroup, def:groups:coset-rel,
    prop:groups:equivalence-coset-rel}
  \lean{Algebra0Lean.Groups.quotientGroup, Algebra0Lean.Groups.quotientMap}
  \leanok
  Let $N$ be a normal subgroup of $G$. The \emph{quotient group of $G$
  modulo $N$}, denoted $G/N$, is the group of equivalence classes of
  $\sim$, with $[a] \bullet [b] := [ab]$. The \emph{quotient map}
  $\pi : G \to G/N$ sends $g$ to $[g]$.
\end{definition}

\begin{proposition}
  \label{prop:groups:isGroupHom-quotientMap}
  \uses{def:groups:quotient-group-modulo-n, def:groups:homomorphism}
  \lean{Algebra0Lean.Groups.isGroupHom_quotientMap}
  The quotient map $\pi : G \to G/N$ is a group homomorphism.
\end{proposition}

\begin{theorem}
  \label{thm:groups:quotient-universal-property}
  \uses{def:groups:quotient-group-modulo-n, def:groups:kernel}
  \lean{Algebra0Lean.Groups.exists_unique_quotient_universal_property}
  Let $N$ be a normal subgroup of $G$. Then for every group
  homomorphism $\varphi : G \to G'$ such that $N \subseteq
  \ker\varphi$, there exists a unique group homomorphism $\bar\varphi
  : G/N \to G'$ such that $\bar\varphi \circ \pi = \varphi$.
\end{theorem}

\subsection{\texorpdfstring{kernel $\Leftrightarrow$ normal}{kernel iff normal}}
\label{subsec:groups:kernel-iff-normal}

\begin{proposition}
  \label{prop:groups:ker-quotientMap}
  \uses{def:groups:quotient-group-modulo-n, def:groups:kernel}
  \lean{Algebra0Lean.Groups.ker_quotientMap}
  The kernel of the quotient map $\pi : G \to G/N$ is $N$ itself.
\end{proposition}

\section{Canonical decomposition and Lagrange's theorem}
\label{sec:groups:canonical-decomposition-and-lagrange-s-theorem}

\subsection{Canonical decomposition}
\label{subsec:groups:canonical-decomposition}

\begin{theorem}
  \label{thm:groups:canonical-decomposition}
  \uses{def:groups:quotient-group-modulo-n, def:groups:kernel,
    def:groups:image, def:groups:subgroup-to-group,
    def:groups:isomorphic-groups}
  \lean{Algebra0Lean.Groups.isomorphicGroups_quotient_ker_image}
  Every group homomorphism $\varphi : G \to G'$ factors as $G
  \twoheadrightarrow G/\ker\varphi \xrightarrow{\sim}
  \operatorname{im}\varphi \hookrightarrow G'$, where the middle map
  is an isomorphism.
\end{theorem}
\begin{proof}
  \uses{thm:groups:quotient-universal-property,
    prop:groups:iso-iff-bijection}
\end{proof}

\begin{corollary}
  \label{coro:groups:first-isomorphism-theorem}
  \uses{thm:groups:canonical-decomposition}
  \lean{Algebra0Lean.Groups.isomorphicGroups_of_surjective_groupHom}
  If $\varphi : G \to G'$ is a surjective group homomorphism, then
  $G' \cong G/\ker\varphi$.
\end{corollary}

\begin{proposition}
  \label{prop:groups:quotient-of-product}
  \uses{def:groups:normal-subgroup, def:groups:quotient-group-modulo-n,
    def:groups:prod-group}
  \lean{Algebra0Lean.Groups.isNormalSubgroup_prod,
    Algebra0Lean.Groups.isomorphicGroups_quotient_prod}
  If $H_1 \subseteq G_1$ and $H_2 \subseteq G_2$ are normal subgroups,
  then $H_1 \times H_2$ is a normal subgroup of $G_1 \times G_2$ and
  $(G_1 \times G_2)/(H_1 \times H_2) \cong (G_1/H_1) \times
  (G_2/H_2)$.
\end{proposition}
\begin{proof}
  \uses{coro:groups:first-isomorphism-theorem}
\end{proof}

\subsection{The index and Lagrange's theorem}
\label{subsec:groups:the-index-and-lagrange-s-theorem}

\begin{definition}
  \label{def:groups:index}
  \uses{def:groups:cosets}
  \lean{Algebra0Lean.Groups.index}
  \leanok
  The \emph{index} of $H$ in $G$, denoted $[G:H]$, is the number of
  left-cosets of $H$ in $G$ (zero if this is infinite).
\end{definition}

\begin{lemma}
  \label{lem:groups:coset-bijections}
  \uses{def:groups:cosets}
  \lean{Algebra0Lean.Groups.bijective_leftCoset_of_mem}
  Let $H$ be a subgroup of a group $G$. Then for every $g \in G$, the
  function $H \to gH$, $h \mapsto gh$, is a bijection.
\end{lemma}

\begin{corollary}[Lagrange's theorem]
  \label{coro:groups:lagrange}
  \uses{def:groups:index, lem:groups:coset-bijections}
  \lean{Algebra0Lean.Groups.card_eq_index_mul_card_subgroup}
  If $G$ is a finite group and $H \subseteq G$ is a subgroup, then
  $|G| = [G:H] \cdot |H|$. In particular, $|H|$ is a divisor of
  $|G|$.
\end{corollary}
