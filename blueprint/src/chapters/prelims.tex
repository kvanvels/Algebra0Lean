\subsection{Equivalence relations, partitions, quotients}

\begin{definition}[Reflexive relation]
  \label{def:prelims:is-reflexive}
  \lean{Algebra0Lean.Prelims.IsReflexive}
  \leanok
  A relation $\sim$ on a set $X$ is \emph{reflexive} if $a \sim a$ for
  every $a \in X$.
\end{definition}

\begin{definition}[Symmetric relation]
  \label{def:prelims:is-symmetric}
  \lean{Algebra0Lean.Prelims.IsSymmetric}
  \leanok
  A relation $\sim$ on a set $X$ is \emph{symmetric} if $a \sim b$
  implies $b \sim a$, for all $a, b \in X$.
\end{definition}

\begin{definition}[Transitive relation]
  \label{def:prelims:is-transitive}
  \lean{Algebra0Lean.Prelims.IsTransitive}
  \leanok
  A relation $\sim$ on a set $X$ is \emph{transitive} if $a \sim b$ and
  $b \sim c$ together imply $a \sim c$, for all $a, b, c \in X$.
\end{definition}

\begin{definition}[Partition]
  \label{def:prelims:is-partition}
  \lean{Algebra0Lean.Prelims.IsPartition}
  \leanok
  A \emph{partition} of a set $X$ is a set $c$ of subsets of $X$ such
  that $\emptyset \notin c$ and every element of $X$ lies in exactly
  one member of $c$.
\end{definition}

\begin{proposition}
  \label{prop:prelims:partition-eq-or-disjoint}
  \uses{def:prelims:is-partition}
  \lean{Algebra0Lean.Prelims.eq_or_disjoint_of_isPartition}
  \leanok
  Any two members of a partition $c$ of $X$ are either equal or
  disjoint.
\end{proposition}
\begin{proof}
  \leanok
\end{proof}

\begin{proposition}
  \label{prop:prelims:partition-not-disjoint-iff-eq}

  \lean{Algebra0Lean.Prelims.not_disjoint_iff_eq_of_isPartition}
  \leanok
  For $s, t \in c$: $s$ and $t$ overlap if and only if $s = t$.

\begin{definition}[Graph of a function]
  \label{def:prelims:graph-of}
  \lean{Algebra0Lean.Prelims.graphOf}
  \leanok
  The graph of $f : X \to Y$: the subset
  $\Gamma_f = \{(a,b) \in X \times Y \mid b = f(a)\} \subseteq X \times Y$.
\end{definition}

\begin{definition}[Graph condition]
  \label{def:prelims:is-graph}
  \lean{Algebra0Lean.Prelims.IsGraph}
  \leanok
  A subset $\Gamma \subseteq X \times Y$ is (the graph of) a function
  $X \to Y$ if every $a \in X$ has exactly one $b \in Y$ with
  $(a,b) \in \Gamma$.
\end{definition}

\begin{proposition}
  \label{prop:prelims:is-graph-graph-of}
  \uses{def:prelims:graph-of, def:prelims:is-graph}
  \lean{Algebra0Lean.Prelims.isGraph_graphOf}
  \leanok
  The graph of a function satisfies the graph condition.
\end{proposition}
\begin{proof}
  \leanok
\end{proof}

\begin{proposition}[A function really "is" its graph]
  \label{prop:prelims:bijective-graph-of}
  \uses{def:prelims:graph-of, def:prelims:is-graph,
    prop:prelims:is-graph-graph-of}
  \lean{Algebra0Lean.Prelims.bijective_graphOf}
  Taking the graph is a bijection between functions $X \to Y$ and
  subsets of $X \times Y$ satisfying the graph condition.
\end{proposition}

\subsection{Injections, surjections, bijections: second viewpoint}
\label{subsec:prelims:injections-surjections-bijections-second-viewpoint}

\begin{proposition}[I.2.4]
  \label{prop:prelims:inv-inj-sur}
  \lean{Algebra0Lean.Prelims.hasLeftInverse_iff_injective,
        Algebra0Lean.Prelims.hasRightInverse_iff_surjective}
  \leanok
  Assume $X \neq \emptyset$, and let $f : X \to Y$ be a function.
  Then $f$ has a left-inverse if and only if it is injective, and $f$
  has a right-inverse if and only if it is surjective.
\end{proposition}
\begin{proof}
  \leanok
\end{proof}

\begin{corollary}[I.2.5]

  \lean{Algebra0Lean.Prelims.bijective_iff_hasInverse}
  \leanok
  A function $f : X \to Y$ is a bijection if and only if it has a
  (two-sided) inverse.
\end{corollary}
\begin{proof}
  \uses{prop:prelims:inv-inj-sur}

\end{proof}

\begin{proposition}
  \label{prop:prelims:no-right-inverse-of-injective-not-surjective}
  \lean{Algebra0Lean.Prelims.not_hasRightInverse_of_injective_not_surjective}

  \label{prop:prelims:many-left-inverses}

\begin{definition}[Section]
  \label{def:prelims:is-section}
  \lean{Algebra0Lean.Prelims.IsSection}

\begin{definition}[Fiber]
  \label{def:prelims:fiber}
  \lean{Algebra0Lean.Prelims.fiber}
  \leanok
  The fiber of $f$ over $q \in Y$: the subset
  $f^{-1}(q) = \{a \in X \mid f(a) = q\} \subseteq X$.

\begin{proposition}

  \leanok
  $f$ is injective if and only if every fiber of $f$ has at most one

  \leanok

\subsection{Monomorphisms and epimorphisms}
\label{subsec:prelims:functions:monomorphisms-and-epimorphisms}

\begin{definition}[Monomorphism]
  \label{def:prelims:monomorphic}
  \lean{Algebra0Lean.Prelims.Monomorphic}
  \leanok
  A function $f : X \to Y$ is a \emph{monomorphism} if for every set
  $Z$ and all functions $\alpha', \alpha'' : Z \to X$,
  $f \circ \alpha' = f \circ \alpha''$ implies $\alpha' = \alpha''$.
\end{definition}

\begin{definition}[Epimorphism]
  \label{def:prelims:epimorphic}
  \lean{Algebra0Lean.Prelims.Epimorphic}
  \leanok
  A function $f : X \to Y$ is an \emph{epimorphism} if every function
  $g : Z \to Y$ factors as $g = f \circ h$ for some $h : Z \to X$.
\end{definition}

\begin{proposition}[I.2.3]
  \label{prop:prelims:2.3}
  \uses{def:prelims:monomorphic}
  \lean{Algebra0Lean.Prelims.injective_iff_monomorphic}
  \leanok
  A function is injective if and only if it is a monomorphism.
\end{proposition}
\begin{proof}
  \leanok
\end{proof}

\begin{exercise}[I.2.5]
  \label{exer:prelims:2.5}
  \uses{def:prelims:epimorphic}
  \lean{Algebra0Lean.Prelims.epimorphic_iff_surjective}
  \leanok
  A function is surjective if and only if it is an epimorphism.
\end{exercise}
\begin{proof}
  \leanok
\end{proof}

\subsection{Canonical decomposition}
\label{subsec:prelims:canonical-decomposition}

\begin{definition}[Kernel pair]
  \label{def:prelims:kernel-pair-setoid}
  \lean{Algebra0Lean.Prelims.kernelPairSetoid}
  \leanok
  The equivalence relation on $X$ induced by a function $f : X \to Y$,
  identifying $x, x'$ whenever $f(x) = f(x')$.
\end{definition}

\begin{definition}[Canonical projection]
  \label{def:prelims:canonical-projection}
  \uses{def:prelims:kernel-pair-setoid}
  \lean{Algebra0Lean.Prelims.canonicalProjection}
  \leanok
  The canonical surjection $X \twoheadrightarrow X/{\sim}$ onto the
  quotient by the kernel pair of $f$.
\end{definition}

\begin{definition}[Canonical bijection]
  \label{def:prelims:canonical-bijection}
  \uses{def:prelims:kernel-pair-setoid}
  \lean{Algebra0Lean.Prelims.canonicalBijection}
  \leanok
  The induced bijection $X/{\sim} \to \image f$.
\end{definition}

\begin{definition}[Canonical inclusion]
  \label{def:prelims:canonical-inclusion}
  \lean{Algebra0Lean.Prelims.canonicalInclusion}
  \leanok
  The inclusion $\image f \hookrightarrow Y$.
\end{definition}

\begin{theorem}[I.2.8, canonical decomposition]
  \label{thm:prelims:canonical-decomposition}
  \uses{def:prelims:canonical-projection, def:prelims:canonical-bijection,
    def:prelims:canonical-inclusion}
  \lean{Algebra0Lean.Prelims.canonicalDecomposition}
  \leanok
  Every function $f : X \to Y$ factors as
  \[
    f = (\text{inclusion}) \circ (\text{bijection}) \circ (\text{projection}),
  \]
  i.e.\ as a surjection, followed by a bijection, followed by an
  injection.
\end{theorem}
\begin{proof}
  \leanok
\end{proof}

\section{Categories}
\label{sec:prelims:categories}

\subsection{Definition}

\begin{definition}[Category]
  \label{def:prelims:category}
  \lean{Algebra0Lean.Prelims.Category}
  \leanok
  A category $\categ{C}$: a type of objects, a type of morphisms
  between each pair of objects, identities, and an associative,
  unital composition law.
\end{definition}

\begin{definition}[Small category]
  \label{def:prelims:category-is-small}
  \uses{def:prelims:category}
  \lean{Algebra0Lean.Prelims.Category.IsSmall}
  \leanok
  A category is small (relative to a universe) if its class of
  objects is actually (equivalent to) a set.
\end{definition}

\begin{definition}[Endomorphism]
  \label{def:prelims:category-end}
  \uses{def:prelims:category}
  \lean{Algebra0Lean.Prelims.Category.End}
  \leanok
  The endomorphisms $\End_{\categ{C}}(A)$ of an object $A$: morphisms
  from $A$ to itself.
\end{definition}

\subsection{Examples}

\begin{definition}[The category of types]
  \label{def:prelims:type-category}
  \uses{def:prelims:category}
  \lean{Algebra0Lean.Prelims.typeCategory}
  \leanok

\end{definition}

\begin{definition}[The category induced by a preorder]

  \lean{Algebra0Lean.Prelims.preorderCategory}
  \leanok

\section{Morphisms}
\label{sec:prelims:morphisms}

\subsection{Isomorphisms}
\label{subsec:prelims:isomorphisms}

\begin{definition}[Isomorphism]
  \label{def:prelims:category-is-iso}
  \uses{def:prelims:category}
  \lean{Algebra0Lean.Prelims.Category.IsIso}
  \leanok
  A morphism $f \in \Hom_{\categ{C}}(A,B)$ is an isomorphism if it has
  a two-sided inverse under composition.
\end{definition}

\begin{proposition}[The inverse of an isomorphism is unique]
  \label{prop:prelims:iso-unique-inverse}
  \uses{def:prelims:category-is-iso}
  \lean{Algebra0Lean.Prelims.Category.isIso_inverse_unique}
  If $g_1$, $g_2 : B \to A$ both act as inverses of an isomorphism
  $f : A \to B$, then $g_1 = g_2$.
\end{proposition}

\begin{proposition}
  \label{prop:prelims:iso-of-left-right-inverse}
  \uses{def:prelims:category-is-iso}
  \lean{Algebra0Lean.Prelims.Category.isIso_of_hasLeftInverse_hasRightInverse}
  If $f$ has a left-inverse $g_1$ and a right-inverse $g_2$, then $f$
  is an isomorphism, $g_1 = g_2$, and this common morphism is the
  (unique) inverse of $f$.
\end{proposition}

\begin{definition}[Inverse of an isomorphism]
  \label{def:prelims:category-iso-inv}
  \uses{def:prelims:category-is-iso}
  \lean{Algebra0Lean.Prelims.Category.IsIso.inv}
  \leanok
  The inverse $f^{-1}$ of an isomorphism $f$.
\end{definition}

\begin{proposition}[Iso properties]
  \label{prop:prelims:iso-props}
  \uses{def:prelims:category-is-iso, def:prelims:category-iso-inv}
  \lean{Algebra0Lean.Prelims.Category.isIso_id,
        Algebra0Lean.Prelims.Category.IsIso.isIso_inv,
        Algebra0Lean.Prelims.Category.IsIso.isIso_comp,
        Algebra0Lean.Prelims.Category.inv_id,
        Algebra0Lean.Prelims.Category.IsIso.inv_inv,
        Algebra0Lean.Prelims.Category.IsIso.inv_comp}
  Each identity $1_A$ is an isomorphism and is its own inverse; if
  $f$ is an isomorphism, then $f^{-1}$ is an isomorphism and
  ${(f^{-1})}^{-1} = f$; and if $f$, $g$ are isomorphisms, then
  $g \circ f$ is an isomorphism and
  ${(g \circ f)}^{-1} = f^{-1} \circ g^{-1}$.
\end{proposition}

\begin{definition}[Isomorphic objects]
  \label{def:prelims:obj-iso}
  \uses{def:prelims:category-is-iso}
  \lean{Algebra0Lean.Prelims.Category.ObjIso}
  \leanok
  Two objects of a category are isomorphic if there is an isomorphism
  between them.
\end{definition}

\begin{proposition}[Isomorphism of objects is an equivalence relation]
  \label{prop:prelims:obj-iso-equiv-rel}
  \uses{def:prelims:obj-iso, prop:prelims:iso-props}
  \lean{Algebra0Lean.Prelims.Category.objIso_refl,
        Algebra0Lean.Prelims.Category.objIso_symm,
        Algebra0Lean.Prelims.Category.objIso_trans}
  An immediate corollary of the iso properties above.
\end{proposition}

\begin{definition}[Groupoid]
  \label{def:prelims:is-groupoid}
  \uses{def:prelims:category-is-iso}
  \lean{Algebra0Lean.Prelims.Category.IsGroupoid}
  \leanok
  A category is a groupoid if every morphism is an isomorphism.
\end{definition}

\begin{definition}[Automorphism]
  \label{def:prelims:aut}
  \uses{def:prelims:category-is-iso, def:prelims:category-end}
  \lean{Algebra0Lean.Prelims.Category.Aut}
  \leanok
  An automorphism of $A$: an isomorphism from $A$ to itself. The set
  of automorphisms $\Aut_{\categ{C}}(A)$, with composition, is a
  group.
\end{definition}

\subsection{Monomorphisms and epimorphisms}
\label{subsec:prelims:morphisms:monomorphisms-and-epimorphisms}

\begin{definition}[Monomorphism]
  \label{def:prelims:category-is-mono}
  \uses{def:prelims:category}
  \lean{Algebra0Lean.Prelims.Category.IsMono}
  \leanok
  A morphism $f : A \to B$ is a monomorphism if it is
  left-cancellable when precomposed with any pair of morphisms into
  $A$.
\end{definition}

\begin{definition}[Epimorphism]
  \label{def:prelims:category-is-epi}
  \uses{def:prelims:category}
  \lean{Algebra0Lean.Prelims.Category.IsEpi}
  \leanok
  A morphism $f : A \to B$ is an epimorphism if it is
  right-cancellable when postcomposed with any pair of morphisms out
  of $B$.
\end{definition}

\begin{proposition}
  \label{prop:prelims:mono-epi-preorder}
  \uses{def:prelims:category-is-mono, def:prelims:category-is-epi,
    def:prelims:preorder-category}
  \lean{Algebra0Lean.Prelims.isMono_and_isEpi_preorderCategory}
  In a category induced by a preorder, every morphism is both a
  monomorphism and an epimorphism.
\end{proposition}

\begin{proposition}
  \label{prop:prelims:iso-iff-bijective-type-category}
  \uses{def:prelims:category-is-iso, def:prelims:type-category}
  \lean{Algebra0Lean.Prelims.isIso_iff_bijective_typeCategory}
  In the category of types, a function is an isomorphism iff it is
  both injective and surjective, i.e.\ iff it is both a monomorphism
  and an epimorphism.
\end{proposition}

\section{Universal properties}
\label{sec:prelims:universal-properties}

\subsection{Initial and final objects}
\label{subsec:prelims:initial-and-final-objects}

\begin{definition}[Initial and final objects]
  \label{def:prelims:initial-final}
  \uses{def:prelims:category}
  \lean{Algebra0Lean.Prelims.Category.IsInitial,
        Algebra0Lean.Prelims.Category.IsFinal}
  \leanok
  $I$ is initial in $\categ{C}$ if for every object $A$ there is
  exactly one morphism $I \to A$; $F$ is final if for every object
  $A$ there is exactly one morphism $A \to F$.
\end{definition}

\begin{proposition}
  \label{prop:prelims:no-initial-final-le-int}
  \uses{def:prelims:initial-final, def:prelims:le-category}
  \lean{Algebra0Lean.Prelims.not_exists_isInitial_leCategory_int,
        Algebra0Lean.Prelims.not_exists_isFinal_leCategory_int}
  The category from $(\Z,\leq)$ has neither an initial nor a final
  object.
\end{proposition}

\begin{proposition}
  \label{prop:prelims:final-slice-le-int}
  \uses{def:prelims:initial-final, def:prelims:slice-category,
    def:prelims:le-category}
  \lean{Algebra0Lean.Prelims.exists_isFinal_sliceCategory_leCategory_int}
  By contrast, the slice category of $(\Z,\leq)$ under $3$ does have
  a final object, namely $(3,3)$; it still has no initial object.
\end{proposition}

\begin{proposition}
  \label{prop:prelims:initial-final-type-category}
  \uses{def:prelims:initial-final, def:prelims:type-category}
  \lean{Algebra0Lean.Prelims.isInitial_typeCategory_iff,
        Algebra0Lean.Prelims.isFinal_typeCategory_unit}
  $\emptyset$ is initial in $\categ{Set}$ (and is the unique initial
  object up to isomorphism); every singleton is final in
  $\categ{Set}$ (so final objects are not unique).
\end{proposition}

\begin{proposition}[Initial/final objects are unique up to a unique isomorphism]
  \label{prop:prelims:initial-final-unique}
  \uses{def:prelims:initial-final, def:prelims:category-is-iso}
  \lean{Algebra0Lean.Prelims.Category.isIso_of_isInitial_isInitial,
        Algebra0Lean.Prelims.Category.isIso_of_isFinal_isFinal}
  If $I_1$, $I_2$ are both initial in $\categ{C}$, then
  $I_1 \cong I_2$; likewise for final objects. Further, these
  isomorphisms are uniquely determined.
\end{proposition}

\subsection{Quotients}
\label{subsec:prelims:quotients}

\begin{definition}[Category of maps respecting an equivalence relation]
  \label{def:prelims:quotient-map-category}
  \uses{def:prelims:category}
  \lean{Algebra0Lean.Prelims.quotientMapCategory}
  \leanok
  The category of maps out of $A$ respecting an equivalence relation
  $\sim$: objects are pairs $(Z,\varphi : A \to Z)$ with $\varphi$
  constant on $\sim$-classes.
\end{definition}

\begin{proposition}[Quotients satisfy a universal property]
  \label{prop:prelims:quotient-universal}
  \uses{def:prelims:quotient-map-category, def:prelims:initial-final}
  \lean{Algebra0Lean.Prelims.isInitial_quotientMapCategory}
  Denoting by $\pi$ the canonical projection, the pair
  $(\pi, A/{\sim})$ is an initial object of this category.
\end{proposition}

\subsection{Products}
\label{subsec:prelims:products}

\begin{definition}[Has products]
  \label{def:prelims:has-products}
  \uses{def:prelims:two-leg-category, def:prelims:initial-final}
  \lean{Algebra0Lean.Prelims.Category.HasProducts}
  \leanok
  A category has (binary) products if $\categ{C}_{A,B}$ has a final
  object, for every $A$, $B$.
\end{definition}

\begin{proposition}
  \label{prop:prelims:has-products-type-category}
  \uses{def:prelims:has-products, def:prelims:type-category}
  \lean{Algebra0Lean.Prelims.hasProducts_typeCategory}
  Products of sets (with their projections) are final objects of
  $\categ{Set}_{A,B}$: the category of types has products.
\end{proposition}

\begin{proposition}
  \label{prop:prelims:has-products-le-int}
  \uses{def:prelims:has-products, def:prelims:le-category}
  \lean{Algebra0Lean.Prelims.hasProducts_leCategory_int}
  The category from $(\Z,\leq)$ has products: the product of $a$ and
  $b$ is $\min(a,b)$.
\end{proposition}
